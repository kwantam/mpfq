MACRO(COPY_FROM_SOURCE filepath)
ADD_CUSTOM_COMMAND(OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/${filepath}
    COMMAND
    cp ${CMAKE_CURRENT_SOURCE_DIR}/${filepath} ${CMAKE_CURRENT_BINARY_DIR}/${filepath}
    DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/${filepath}
    VERBATIM)
ENDMACRO(COPY_FROM_SOURCE)

MACRO(COPY_FROM_SOURCE_TO_INC filepath)
ADD_CUSTOM_COMMAND(OUTPUT ${MPFQ_INC_DIR}/${filepath}
    COMMAND
    cp ${CMAKE_CURRENT_SOURCE_DIR}/${filepath} ${MPFQ_INC_DIR}/${filepath}
    DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/${filepath}
    VERBATIM)
ENDMACRO(COPY_FROM_SOURCE_TO_INC)
